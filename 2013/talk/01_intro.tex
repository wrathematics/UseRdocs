\begin{frame}[fragile]
\begin{block}{pbdMPI vs Rmpi}
\pause
\begin{lstlisting}[title=Reduce Operation with Rmpi]
# int
mpi.allreduce(x, type=1)
# double
mpi.allreduce(x, type=2)
\end{lstlisting}

\begin{lstlisting}[title=Reduce Operation with pbdMPI]
allreduce(x)
\end{lstlisting}

\pause
